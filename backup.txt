private void classe()
    {
        if(isToken(currentToken,TokenType.CLASS))
        {
            currentToken = nextToken();
            if(isToken(currentToken,TokenType.TYPE_IDENTIFIER))
            {
                currentToken = nextToken();
                if(isToken(currentToken,TokenType.INHERITS))
                {
                    currentToken = nextToken();
                    if(isToken(currentToken,TokenType.TYPE_IDENTIFIER)) currentToken = nextToken();
                    else createError("tipo de classe");
                }
                
                if(isToken(currentToken,TokenType.OPEN_BRACES))
                {
                    feature();
                    currentToken = nextToken();
                    if(!isToken(currentToken,TokenType.CLOSE_BRACES)) createError("}");
                }
                else createError("{");
            }
            else createError("tipo de classe");
        }
        else createError("class");
    }
